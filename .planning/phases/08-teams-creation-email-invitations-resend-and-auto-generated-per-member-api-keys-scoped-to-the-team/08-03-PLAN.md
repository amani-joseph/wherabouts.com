  <behavior>
    - encryptSecret(plaintext): returns `{ ciphertext: string, iv: string }` (both hex-encoded).
    - Encryption uses AES-256-GCM with a 12-byte IV generated from `crypto.randomBytes(12)`.
    - The KEY is `Buffer.from(serverEnv.KEY_ENC_KEY, "hex")` (32 bytes).
    - Ciphertext is `authTag.toString("hex") + ":" + encrypted.toString("hex")` so auth tag travels with ciphertext.
    - decryptSecret({ ciphertext, iv }): reverses the above, returning the original plaintext string.
    - decryptSecret throws if the auth tag fails (tamper detection).
    - Test 1: encrypt then decrypt returns identical plaintext.
    - Test 2: two calls to encryptSecret on same plaintext produce different ciphertext+iv (IV randomness).
    - Test 3: mutating the ciphertext hex string causes decryptSecret to throw.
  </behavior>
  <action>
    1. Create `apps/web/src/lib/api-key-crypto.ts`:
       ```ts
       import { createCipheriv, createDecipheriv, randomBytes } from "node:crypto";
       import { serverEnv } from "@wherabouts.com/env/server";

       const ALGORITHM = "aes-256-gcm" as const;
       const IV_LENGTH = 12;
       const AUTH_TAG_LENGTH = 16;

       function getKey(): Buffer {
         return Buffer.from(serverEnv.KEY_ENC_KEY, "hex");
       }

       export function encryptSecret(plaintext: string): { ciphertext: string; iv: string } {
         const iv = randomBytes(IV_LENGTH);
         const cipher = createCipheriv(ALGORITHM, getKey(), iv, { authTagLength: AUTH_TAG_LENGTH });
         const encrypted = Buffer.concat([cipher.update(plaintext, "utf8"), cipher.final()]);
         const authTag = cipher.getAuthTag();
         return {
           ciphertext: `${authTag.toString("hex")}:${encrypted.toString("hex")}`,
           iv: iv.toString("hex"),
         };
       }

       export function decryptSecret({ ciphertext, iv }: { ciphertext: string; iv: string }): string {
         const [authTagHex, encryptedHex] = ciphertext.split(":");
         if (!authTagHex || !encryptedHex) {
           throw new Error("Invalid ciphertext format");
         }
         const decipher = createDecipheriv(ALGORITHM, getKey(), Buffer.from(iv, "hex"), { authTagLength: AUTH_TAG_LENGTH });
         decipher.setAuthTag(Buffer.from(authTagHex, "hex"));
         const decrypted = Buffer.concat([
           decipher.update(Buffer.from(encryptedHex, "hex")),
           decipher.final(),
         ]);
         return decrypted.toString("utf8");
       }
       ```
