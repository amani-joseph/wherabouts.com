    ```tsx
    import { createFileRoute, useNavigate, Link } from "@tanstack/react-router";
    import { useEffect, useState } from "react";
    import { authClient, useSession } from "@/lib/auth-client.ts";
    import { OneTimeKeyModal } from "@/components/one-time-key-modal.tsx";
    import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card.tsx";
    import { Button } from "@/components/ui/button.tsx";

    export const Route = createFileRoute("/invite/$id")({
      component: InviteAcceptPage,
      loader: async ({ params }) => {
        const { data } = await authClient.organization.getInvitation({ query: { id: params.id } });
        return { invitation: data ?? null };
      },
    });

    function InviteAcceptPage() {
      const { id } = Route.useParams();
      const { invitation } = Route.useLoaderData();
      const { data: session } = useSession();
      const navigate = useNavigate();
      const [plaintextKey, setPlaintextKey] = useState<string | null>(null);
      const [status, setStatus] = useState<"idle" | "accepting" | "accepted" | "error" | "expired">("idle");

      useEffect(() => {
        if (!session || !invitation) return;
        if (status !== "idle") return;
        setStatus("accepting");
        (async () => {
          try {
            const res = await authClient.organization.acceptInvitation({ invitationId: id });
            // The afterAcceptInvitation hook generated the key server-side.
            // Fetch the freshly-generated plaintext via a one-time query-param channel
            // OR rely on the server fn returning it directly via res.data.key.
            // If BetterAuth surfaces it via res.data, use that; otherwise we cannot
            // reveal plaintext later (encrypted reveal via revealApiKeySecret is fine
            // but the UI-SPEC one-time modal wants plaintext NOW).
            // Redirect with ?newKey=<id>; /team will call revealApiKeySecret to show it.
            const newKeyId = (res as { data?: { apiKey?: { id?: string } } }).data?.apiKey?.id;
            if (newKeyId) {
              navigate({ to: "/team", search: { newKey: newKeyId } });
            } else {
              navigate({ to: "/team" });
            }
          } catch (err) {
            const msg = (err as Error).message ?? "";
            setStatus(/expired|already/i.test(msg) ? "expired" : "error");
          }
        })();
      }, [session, invitation, id, status, navigate]);

      if (!invitation) {
        return (
          <CenteredCard>
            <CardTitle className="text-2xl font-semibold tracking-tight">This invitation has expired or has already been used.</CardTitle>
            <Button asChild><Link to="/team">Go to team</Link></Button>
          </CenteredCard>
        );
      }

      if (!session) {
        return (
          <CenteredCard>
            <CardHeader>
              <CardTitle className="text-2xl font-semibold tracking-tight">You've been invited to join {invitation.organizationName}</CardTitle>
              <CardDescription className="text-sm">{invitation.inviterName} has invited you to join their team.</CardDescription>
            </CardHeader>
            <CardContent className="flex flex-col gap-2">
              <Button asChild><Link to="/sign-in" search={{ redirect: `/invite/${id}` }}>Sign in to accept</Link></Button>
              <Link to="/sign-up" search={{ redirect: `/invite/${id}` }} className="text-xs text-muted-foreground">Don't have an account? Create one</Link>
            </CardContent>
          </CenteredCard>
        );
      }

      if (status === "expired") {
        return (
          <CenteredCard>
            <CardTitle className="text-2xl font-semibold tracking-tight">This invitation has expired or has already been used.</CardTitle>
            <Button asChild><Link to="/team">Go to team</Link></Button>
          </CenteredCard>
        );
      }

      return null; // mid-accept, navigation imminent
    }

    function CenteredCard({ children }: { children: React.ReactNode }) {
      return (
        <div className="flex min-h-screen items-center justify-center p-6">
          <Card className="w-full max-w-md">{children}</Card>
        </div>
      );
    }
    ```

    - Primary CTA: `"Sign in to accept"`
    - Secondary link: `"Don't have an account? Create one"`
    - Expired copy: `"This invitation has expired or has already been used."`
    - Expired button: `"Go to team"`

    Tabs, double quotes, `.ts`/`.tsx` extension in relative imports.

    Note: if `authClient.organization.getInvitation` returns a different field shape than `organizationName`/`inviterName`, adapt to the actual BetterAuth response — consult context7 for `better-auth` organization plugin docs, the field is typically `organization.name` and `inviter.user.name`.
