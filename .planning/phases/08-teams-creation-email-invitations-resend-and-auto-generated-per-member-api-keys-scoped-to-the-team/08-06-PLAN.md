    ```tsx
    import { createFileRoute } from "@tanstack/react-router";
    import { useState, useEffect } from "react";
    import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
    import { z } from "zod";
    import { MoreHorizontal } from "lucide-react";
    import { authClient, useActiveOrganization, useSession } from "@/lib/auth-client.ts";
    import { listTeamMembers, listTeamInvitations, removeMember, revokeInvitation, resendInvitation } from "@/lib/teams-server.ts";
    import { revealApiKeySecret } from "@/lib/api-keys-server.ts";
    import { Card, CardHeader, CardTitle, CardDescription, CardContent } from "@/components/ui/card.tsx";
    import { Button } from "@/components/ui/button.tsx";
    import { Badge } from "@/components/ui/badge.tsx";
    import { Avatar, AvatarFallback } from "@/components/ui/avatar.tsx";
    import { DropdownMenu, DropdownMenuContent, DropdownMenuItem, DropdownMenuTrigger } from "@/components/ui/dropdown-menu.tsx";
    import { InviteMemberDialog } from "@/components/invite-member-dialog.tsx";
    import { RemoveMemberAlert } from "@/components/remove-member-alert.tsx";
    import { OneTimeKeyModal } from "@/components/one-time-key-modal.tsx";
    import { toast } from "sonner";

    const searchSchema = z.object({ newKey: z.string().uuid().optional() });

    export const Route = createFileRoute("/_protected/team")({
      component: TeamPage,
      validateSearch: searchSchema,
    });

    function TeamPage() {
      const { data: session } = useSession();
      const { data: activeOrg } = useActiveOrganization();
      const { newKey } = Route.useSearch();
      const [inviteOpen, setInviteOpen] = useState(false);
      const [removeTarget, setRemoveTarget] = useState<{ id: string; name: string } | null>(null);
      const [revealedKey, setRevealedKey] = useState<string | null>(null);
      const qc = useQueryClient();

      const teamId = activeOrg?.id;
      const currentUserRole = /* determine from activeOrg.members or fetch separately */ "member";

      const members = useQuery({
        queryKey: ["team-members", teamId],
        queryFn: () => listTeamMembers({ data: { teamId: teamId! } }),
        enabled: !!teamId,
      });
      const invitations = useQuery({
        queryKey: ["team-invitations", teamId],
        queryFn: () => listTeamInvitations({ data: { teamId: teamId! } }),
        enabled: !!teamId && (currentUserRole === "owner" || currentUserRole === "admin"),
      });

      // Auto-reveal newKey after invite-accept redirect
      useEffect(() => {
        if (!newKey || revealedKey) return;
        (async () => {
          try {
            const res = await revealApiKeySecret({ data: { keyId: newKey } });
            setRevealedKey(res.plaintext);
          } catch {
            // if owner changed or key missing, silently skip
          }
        })();
      }, [newKey, revealedKey]);

      const removeMut = useMutation({
        mutationFn: (memberId: string) => removeMember({ data: { teamId: teamId!, memberId } }),
        onSuccess: () => { qc.invalidateQueries({ queryKey: ["team-members", teamId] }); setRemoveTarget(null); },
      });
      const revokeMut = useMutation({
        mutationFn: (invitationId: string) => revokeInvitation({ data: { invitationId } }),
        onSuccess: () => qc.invalidateQueries({ queryKey: ["team-invitations", teamId] }),
      });
      const resendMut = useMutation({
        mutationFn: (invitationId: string) => resendInvitation({ data: { invitationId } }),
        onSuccess: () => toast("Invitation re-sent"),
      });

      if (!activeOrg) return null;

      return (
        <div className="flex flex-col gap-8 p-6">
          <header className="flex items-center justify-between">
            <div>
              <h1 className="text-2xl font-semibold tracking-tight">Team</h1>
              <p className="text-sm text-muted-foreground">Manage your team members, invitations, and access.</p>
            </div>
            <Button onClick={() => setInviteOpen(true)}>Invite Member</Button>
          </header>

          {/* Members Card */}
          <Card>
            <CardHeader>
              <CardTitle>Members</CardTitle>
              <CardDescription>
                {members.data?.length ?? 0} {members.data?.length === 1 ? "member" : "members"} in {activeOrg.name}
              </CardDescription>
            </CardHeader>
            <CardContent className="divide-y">
              {members.data?.map((m) => (
                <div key={m.id} className="flex items-center gap-4 py-4 first:pt-0 last:pb-0">
                  <Avatar><AvatarFallback>{(m.name ?? m.email).slice(0, 2).toUpperCase()}</AvatarFallback></Avatar>
                  <div className="flex-1 min-w-0">
                    <div className="text-sm font-medium">{m.name ?? m.email}{m.userId === session?.user.id ? " (you)" : ""}</div>
                    <div className="text-xs text-muted-foreground">{m.email}</div>
                  </div>
                  <Badge variant="outline">{m.role}</Badge>
                  {(currentUserRole === "owner" || currentUserRole === "admin") && m.userId !== session?.user.id && (
                    <DropdownMenu>
                      <DropdownMenuTrigger asChild>
                        <Button variant="ghost" size="icon" className="size-8" aria-label="Member actions"><MoreHorizontal className="size-4" /></Button>
                      </DropdownMenuTrigger>
                      <DropdownMenuContent align="end">
                        <DropdownMenuItem onClick={() => setRemoveTarget({ id: m.id, name: m.name ?? m.email })} className="text-destructive">Remove member</DropdownMenuItem>
                      </DropdownMenuContent>
                    </DropdownMenu>
                  )}
                </div>
              ))}
            </CardContent>
          </Card>

          {/* Pending Invitations Card — hidden when empty */}
          {invitations.data && invitations.data.length > 0 && (
            <Card>
              <CardHeader><CardTitle>Pending Invitations</CardTitle></CardHeader>
              <CardContent className="divide-y">
                {invitations.data.map((inv) => (
                  <div key={inv.id} className="flex items-center gap-4 py-4 first:pt-0 last:pb-0">
                    <div className="flex-1 min-w-0">
                      <div className="text-sm">{inv.email}</div>
                      <div className="text-xs text-muted-foreground">
                        Invited {daysAgo(inv.createdAt)} ago as {inv.role}
                      </div>
                    </div>
                    <Badge variant="outline" className="text-amber-600">Pending</Badge>
                    <Button variant="outline" size="sm" onClick={() => resendMut.mutate(inv.id)}>Resend</Button>
                    <Button variant="ghost" size="sm" onClick={() => revokeMut.mutate(inv.id)}>Revoke</Button>
                  </div>
                ))}
              </CardContent>
            </Card>
          )}

          <InviteMemberDialog
            open={inviteOpen}
            onOpenChange={setInviteOpen}
            teamId={teamId!}
            currentUserRole={currentUserRole}
            onInvited={() => qc.invalidateQueries({ queryKey: ["team-invitations", teamId] })}
          />
          {removeTarget && (
            <RemoveMemberAlert
              open={!!removeTarget}
              onOpenChange={(o) => !o && setRemoveTarget(null)}
              memberName={removeTarget.name}
              teamName={activeOrg.name}
              onConfirm={() => removeMut.mutate(removeTarget.id)}
            />
          )}
          {revealedKey && (
            <OneTimeKeyModal
              open={!!revealedKey}
              plaintextKey={revealedKey}
              mode="accept"
              onContinue={() => setRevealedKey(null)}
            />
          )}
        </div>
      );
    }

    function daysAgo(date: Date | string): string {
      const ms = Date.now() - new Date(date).getTime();
      const days = Math.floor(ms / (1000 * 60 * 60 * 24));
      return days === 0 ? "today" : `${days} ${days === 1 ? "day" : "days"}`;
    }
    ```
