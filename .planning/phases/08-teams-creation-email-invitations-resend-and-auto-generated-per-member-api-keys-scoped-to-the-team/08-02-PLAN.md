    2. Create `packages/database/scripts/backfill-personal-teams.ts`:
       ```ts
       import { eq, isNull } from "drizzle-orm";
       import { db } from "../src/db.ts";
       import { users } from "../src/schema/auth.ts";
       import { projects } from "../src/schema/projects.ts";
       import { apiKeys } from "../src/schema/api-keys.ts";
       import { teams, teamMembers } from "../src/schema/teams.ts";

       function slugify(input: string): string {
         return input.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-|-$/g, "") + "-personal";
       }

       export async function backfillPersonalTeams(): Promise<void> {
         const allUsers = await db.select().from(users);
         for (const user of allUsers) {
           const displayName = user.name ?? user.email.split("@")[0];
           const baseSlug = slugify(displayName || user.id);

           // Skip if user already has ≥1 team membership
           const existing = await db.select().from(teamMembers).where(eq(teamMembers.userId, user.id));
           if (existing.length > 0) continue;

           const [team] = await db.insert(teams).values({
             name: `${displayName}'s Personal`,
             slug: `${baseSlug}-${user.id.slice(0, 8)}`,
           }).returning();

           await db.insert(teamMembers).values({
             teamId: team.id,
             userId: user.id,
             role: "owner",
           });

           // Reassign only rows where team_id is still null AND owned by this user
           await db.update(projects).set({ teamId: team.id }).where(
             eq(projects.userId, user.id)
           );
           await db.update(apiKeys).set({ teamId: team.id }).where(
             eq(apiKeys.userId, user.id)
           );
         }
         console.log(`Backfill complete: processed ${allUsers.length} users`);
       }

       backfillPersonalTeams().then(() => process.exit(0)).catch((err) => {
         console.error(err);
         process.exit(1);
       });
       ```
