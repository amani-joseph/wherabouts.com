    import { createServerFn } from "@tanstack/start";
    import { z } from "zod";
    import { and, eq } from "drizzle-orm";
    import { db } from "@wherabouts.com/database";
    import { teams, teamMembers, teamInvitations } from "@wherabouts.com/database/schema";
    import { users } from "@wherabouts.com/database/schema";
    import { getSession } from "./auth-server.ts";
    import { auth } from "@wherabouts.com/auth";

    async function requireSession() {
      const session = await getSession();
      if (!session) throw new Error("Unauthorized");
      return session;
    }

    async function requireRole(teamId: string, userId: string, allowed: Array<"owner" | "admin" | "member">) {
      const [row] = await db.select().from(teamMembers).where(
        and(eq(teamMembers.teamId, teamId), eq(teamMembers.userId, userId))
      );
      if (!row || !allowed.includes(row.role as "owner" | "admin" | "member")) {
        throw new Error("Forbidden");
      }
      return row;
    }

    export const createTeam = createServerFn({ method: "POST" })
      .inputValidator(z.object({ name: z.string().min(1).max(80) }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        // Delegate to BetterAuth organization.create so all its hooks (key gen, active org) run
        return await auth.api.createOrganization({ body: { name: data.name, slug: slugify(data.name) }, headers: await getHeaders() });
      });

    export const listTeamMembers = createServerFn({ method: "GET" })
      .inputValidator(z.object({ teamId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        await requireRole(data.teamId, session.user.id, ["owner", "admin", "member"]);
        return await db.select({
          id: teamMembers.id,
          userId: teamMembers.userId,
          role: teamMembers.role,
          createdAt: teamMembers.createdAt,
          email: users.email,
          name: users.name,
        }).from(teamMembers).innerJoin(users, eq(users.id, teamMembers.userId))
          .where(eq(teamMembers.teamId, data.teamId));
      });

    export const listTeamInvitations = createServerFn({ method: "GET" })
      .inputValidator(z.object({ teamId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        await requireRole(data.teamId, session.user.id, ["owner", "admin"]);
        return await db.select().from(teamInvitations).where(
          and(eq(teamInvitations.teamId, data.teamId), eq(teamInvitations.status, "pending"))
        );
      });

    export const removeMember = createServerFn({ method: "POST" })
      .inputValidator(z.object({ teamId: z.string().uuid(), memberId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        await requireRole(data.teamId, session.user.id, ["owner", "admin"]);
        // Cannot remove yourself or the sole owner — guard
        const target = await db.select().from(teamMembers).where(eq(teamMembers.id, data.memberId));
        if (!target[0] || target[0].teamId !== data.teamId) throw new Error("Not found");
        if (target[0].userId === session.user.id) throw new Error("Cannot remove yourself");
        if (target[0].role === "owner") {
          const owners = await db.select().from(teamMembers).where(and(eq(teamMembers.teamId, data.teamId), eq(teamMembers.role, "owner")));
          if (owners.length === 1) throw new Error("Cannot remove the sole owner");
        }
        await db.delete(teamMembers).where(eq(teamMembers.id, data.memberId));
        return { ok: true };
      });

    export const revokeInvitation = createServerFn({ method: "POST" })
      .inputValidator(z.object({ invitationId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        const [inv] = await db.select().from(teamInvitations).where(eq(teamInvitations.id, data.invitationId));
        if (!inv) throw new Error("Not found");
        await requireRole(inv.teamId, session.user.id, ["owner", "admin"]);
        await db.update(teamInvitations).set({ status: "revoked" }).where(eq(teamInvitations.id, data.invitationId));
        return { ok: true };
      });

    export const resendInvitation = createServerFn({ method: "POST" })
      .inputValidator(z.object({ invitationId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await requireSession();
        const [inv] = await db.select().from(teamInvitations).where(eq(teamInvitations.id, data.invitationId));
        if (!inv) throw new Error("Not found");
        await requireRole(inv.teamId, session.user.id, ["owner", "admin"]);
        // Delegate to BetterAuth so it re-sends through sendInvitationEmail
        return await auth.api.createInvitation({ body: { email: inv.email, organizationId: inv.teamId, role: inv.role }, headers: await getHeaders() });
      });
    ```

    import { createServerFn } from "@tanstack/start";
    import { z } from "zod";
    import { and, eq } from "drizzle-orm";
    import { randomBytes, scryptSync, timingSafeEqual } from "node:crypto";
    import { db } from "@wherabouts.com/database";
    import { apiKeys, teamMembers } from "@wherabouts.com/database/schema";
    import { getSession } from "./auth-server.ts";
    import { encryptSecret, decryptSecret } from "./api-key-crypto.ts";

    const SECRET_BYTES = 32;
    const SALT_BYTES = 16;
    const SCRYPT_KEYLEN = 64;
    const KEY_PREFIX = "whb_";

    function generateSecret(): string {
      return KEY_PREFIX + randomBytes(SECRET_BYTES).toString("base64url");
    }

    function hashSecret(secret: string, salt: Buffer): Buffer {
      return scryptSync(secret, salt, SCRYPT_KEYLEN);
    }

    async function requireMembership(teamId: string, userId: string, allowed: Array<"owner" | "admin" | "member">) {
      const [row] = await db.select().from(teamMembers).where(
        and(eq(teamMembers.teamId, teamId), eq(teamMembers.userId, userId))
      );
      if (!row || !allowed.includes(row.role as "owner" | "admin" | "member")) throw new Error("Forbidden");
      return row;
    }

    export const listApiKeys = createServerFn({ method: "GET" })
      .inputValidator(z.object({ teamId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await getSession();
        if (!session) throw new Error("Unauthorized");
        await requireMembership(data.teamId, session.user.id, ["owner", "admin", "member"]);
        // Return metadata only — no secrets
        return await db.select({
          id: apiKeys.id,
          name: apiKeys.name,
          userId: apiKeys.userId,
          teamId: apiKeys.teamId,
          projectId: apiKeys.projectId,
          suffix: apiKeys.suffix,
          lastUsedAt: apiKeys.lastUsedAt,
          createdAt: apiKeys.createdAt,
        }).from(apiKeys).where(eq(apiKeys.teamId, data.teamId));
      });

    export const createApiKey = createServerFn({ method: "POST" })
      .inputValidator(z.object({
        teamId: z.string().uuid(),
        name: z.string().min(1).max(80),
        memberUserId: z.string().uuid(), // who the key belongs to
        projectId: z.string().uuid().nullable().optional(), // null = team-scoped
      }))
      .handler(async ({ data }) => {
        const session = await getSession();
        if (!session) throw new Error("Unauthorized");
        // Only owner/admin can create keys for others; members can never self-create (per CONTEXT.md)
        await requireMembership(data.teamId, session.user.id, ["owner", "admin"]);

        const plaintext = generateSecret();
        const salt = randomBytes(SALT_BYTES);
        const hash = hashSecret(plaintext, salt);
        const { ciphertext, iv } = encryptSecret(plaintext);
        const suffix = plaintext.slice(-4);

        const [row] = await db.insert(apiKeys).values({
          name: data.name,
          userId: data.memberUserId,
          teamId: data.teamId,
          projectId: data.projectId ?? null,
          suffix,
          secretHash: hash.toString("hex"),
          secretSalt: salt.toString("hex"),
          secretCiphertext: ciphertext,
          secretIv: iv,
        }).returning();

        return { ...row, plaintext }; // plaintext returned ONCE at creation
      });

    export const revokeApiKey = createServerFn({ method: "POST" })
      .inputValidator(z.object({ keyId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await getSession();
        if (!session) throw new Error("Unauthorized");
        const [key] = await db.select().from(apiKeys).where(eq(apiKeys.id, data.keyId));
        if (!key) throw new Error("Not found");
        await requireMembership(key.teamId, session.user.id, ["owner", "admin"]);
        await db.delete(apiKeys).where(eq(apiKeys.id, data.keyId));
        return { ok: true };
      });

    export const revealApiKeySecret = createServerFn({ method: "POST" })
      .inputValidator(z.object({ keyId: z.string().uuid() }))
      .handler(async ({ data }) => {
        const session = await getSession();
        if (!session) throw new Error("Unauthorized");
        const [key] = await db.select().from(apiKeys).where(eq(apiKeys.id, data.keyId));
        if (!key) throw new Error("Not found");
        if (key.userId !== session.user.id) {
          const err = new Error("Forbidden");
          (err as Error & { status?: number }).status = 403;
          throw err;
        }
        if (!key.secretCiphertext || !key.secretIv) {
          throw new Error("This legacy key cannot be revealed. Create a new key.");
        }
        return { plaintext: decryptSecret({ ciphertext: key.secretCiphertext, iv: key.secretIv }) };
      });
    ```

    export async function generateApiKeyForMember(teamId: string, userId: string): Promise<{ plaintext: string; id: string }> {
      const plaintext = generateSecret();
      const salt = randomBytes(SALT_BYTES);
      const hash = hashSecret(plaintext, salt);
      const { ciphertext, iv } = encryptSecret(plaintext);
      const [row] = await db.insert(apiKeys).values({
        name: "Default team key",
        userId, teamId, projectId: null,
        suffix: plaintext.slice(-4),
        secretHash: hash.toString("hex"),
        secretSalt: salt.toString("hex"),
        secretCiphertext: ciphertext,
        secretIv: iv,
      }).returning();
      return { plaintext, id: row.id };
    }
